The sendto() function takes three parameters: the data to be sent, the length of the data, and the address and port number of the recipient.

  3. On the receiving end, you would use the recvfrom() function to receive data sent to the specified IP address and port number. The recvfrom() function takes two parameters: a buffer to store the received data, and the maximum length of the buffer.

c
#include <iostream>
#include <string>
#include <cstring>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>

using namespace std;

int main() {
    int sockfd, n;
    char buffer[1024];
    struct sockaddr_in servaddr, cliaddr;
    
    // create a UDP socket
    sockfd = socket(AF_INET, SOCK_DGRAM, 0);
    
    // clear the server address structure
    memset(&servaddr, 0, sizeof(servaddr));
    
    // fill in the server address structure
    servaddr.sin_family = AF_INET;
    servaddr.sin_addr.s_addr = INADDR_ANY;
    servaddr.sin_port = htons(5000);
    
    // bind the socket to the server address
    bind(sockfd, (const struct sockaddr *)&servaddr, sizeof(servaddr));
    
    // receive data from client
    socklen_t len = sizeof(cliaddr);
    n = recvfrom(sockfd, (char *)buffer, 1024, MSG_WAITALL, (struct sockaddr *)&cliaddr, &len);
    buffer[n] = '\0';
    
    // print the received message
    cout << "Received message: " << buffer << endl;
    
    // send a response back to the client
    string response = "Hello, client!";
    sendto(sockfd, response.c_str(), response.length(), MSG_CONFIRM, (const struct sockaddr *)&cliaddr, len);
    
    // close the socket
    close(sockfd);
    
    return 0;
}

In this example, we create a UDP socket using socket() and bind it to the server address using bind(). We then use recvfrom() to receive data from the client and sendto() to send a response back to the client. Finally, we close the socket using close(). Note that error checking has been omitted for brevity.
