To scrape data from a website and save it to a database in Python, you can use libraries such as requests for making HTTP requests and BeautifulSoup for parsing HTML. Additionally, you may use a database library like sqlite3 for SQLite or psycopg2 for PostgreSQL, depending on your database choice.

Here's a simple example using requests, BeautifulSoup, and sqlite3 to scrape employee data from the provided URL and save it to an SQLite database: 

python
import requests
from bs4 import BeautifulSoup
import sqlite3

# Function to scrape data from the website
def scrape_and_save_to_db():
    # URL of the website to scrape
    url = "https://techybusinessservices.co.uk/examPython.php"

    # Send a GET request to the URL
    response = requests.get(url)

    if response.status_code == 200:
        # Parse the HTML content using BeautifulSoup
        soup = BeautifulSoup(response.content, 'html.parser')

        # Extract employee data
        employee_data = []
        for row in soup.find_all('tr')[1:]:  # Skip the header row
            columns = row.find_all('td')
            name = columns[0].text.strip()
            email = columns[1].text.strip()
            salary = columns[2].text.strip()
            address = columns[3].text.strip()

            employee_data.append((name, email, salary, address))

        # Save data to SQLite database
        save_to_database(employee_data)
        print("Data scraped and saved to the database.")
    else:
        print("Failed to retrieve data. HTTP Status Code:", response.status_code)
