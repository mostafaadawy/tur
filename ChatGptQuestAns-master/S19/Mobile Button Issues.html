function MyComponent({ isAdmin }) {

      )}
    </div>
  );

In this case, if isAdmin is true, the first <p> will be rendered, and if it's false, the second <p> will be rendered. This is achieved using the ternary conditional operator (? :).

jsx
import { useEffect, useState } from 'react';

function MyComponent() {
  const [windowWidth, setWindowWidth] = useState(window.innerWidth);

  // Update window width on resize
  useEffect(() => {
    const handleResize = () => {
      setWindowWidth(window.innerWidth);
    };

    window.addEventListener('resize', handleResize);

    // Cleanup the event listener on component unmount
    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, []); // Empty dependency array to run the effect only once on mount

  return (
    <div>
      <p>Screen Width: {windowWidth}px</p>
      {/* Your component content goes here */}
    </div>
  );
}

export default MyComponent;

In this example:

  • The useEffect hook is used to add an event listener for the 'resize' event.
  • The window.innerWidth is initially set when the component mounts.
  • The handleResize function updates the windowWidth state whenever the window is resized.
  • The event listener is removed when the component is unmounted to prevent memory leaks.
