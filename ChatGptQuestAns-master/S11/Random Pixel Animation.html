from PIL import Image

def generate_js_file(image_path, js_file_path):
    image = Image.open(image_path)
    width, height = image.size

    pixels = image.load()

    points = []
    for y in range(height):
        for x in range(width):
            r, g, b = pixels[x, y][:3]  # Get RGB values of the pixel
            points.append((x, y, r, g, b))

    with open(js_file_path, 'w') as js_file:
        js_file.write('const pixelPoints = [\n')
        for point in points:
            js_file.write(f'  {point},\n')
        js_file.write('];')

js_file_path = 'output.js'

generate_js_file(image_path, js_file_path)

Make sure to replace 'path_to_your_image.webp' with the actual path to your WebP image file, and 'output.js' with the desired path for the output JavaScript file.
